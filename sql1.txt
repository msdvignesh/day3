CREATE TABLE EMP2(
EMPNO INT CONSTRAINT EMP_PK PRIMARY KEY,
ENAME VARCHAR(20),
GENDER CHAR(1) CONSTRAINT EMP_GENDER DEFAULT 'F',
JOB VARCHAR(10) CONSTRAINT EMP_JOB CHECK (JOB IN('SALESMAN','MANAGER','CLERK','ANALYST','PRESIDENT')),
MGR INT CONSTRAINT EMP_MGR REFERENCES EMP(EMPNO),
HIREDATE DATE CONSTRAINT CHECK (DATE>'2022-04-01'),
SAL INT CONSTRAINT EMP_SAL CHECK (SAL>300),
COMM INT CONSTRAINT EMP_COMM CHECK (COMM>200),
DEPTNO INT CONSTRAINT EMP_DEPTNO REFERENCE DEPT(DEPTNO)
);


CREATE TABLE DEPT(
DEPTNO INT CONSTRAINT DEPT_DEPTNO PRIMARY KEY,
DNAME VARCHAR(10),
DLOC VARCHAR(20));
